function fetchTargetProducts() { document.getElementById("searchBox").placeholder = "Loading Products from CRM..."; log("Attempting Standard Search..."); // STRATEGY CHANGE: // We use .searchRecords() instead of .coql(). It is much more robust. // We fetch ALL products where Hardware_Type is NA, then filter for 'phone' in Javascript. // 1. Define Criteria: ((Hardware_Type:equals:NA)) // Note: Ensure 'Hardware_Type' matches your API name exactly. ZOHO.CRM.API.searchRecord({ Entity: "Products", Type: "criteria", Query: "((Hardware_Type:equals:NA))" }) .then(function(data){ if(data.data){ let rawList = data.data; log("Raw Fetch Complete. Found " + rawList.length + " items. Filtering for 'phone'..."); // 2. Client-Side Filter: Keep only items where Category contains "phone" // Using "Category" API Name we found earlier. masterProductList = rawList.filter(p => p.Category != null && p.Category.toLowerCase().includes("phone") ); let count = masterProductList.length; // 3. Feedback log("FILTER SUCCESS: Found " + count + " phones."); document.getElementById("searchBox").placeholder = "Tap to view " + count + " phones..."; alert("Success! Loaded " + count + " phones."); } else { log("Search ran but returned 0 items."); document.getElementById("searchBox").placeholder = "No products found."; } }) .catch(function(err){ let errText = JSON.stringify(err); log("SEARCH ERROR: " + errText); alert("API Error: " + errText); }); }
